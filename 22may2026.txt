SELECT * FROM 
departments CROSS JOIN hod

SELECT * FROM departments 
INNER JOIN hod 
where departments.did = hod.hid;

SELECT * FROM departments 
NATURAL JOIN hod;

SELECT * FROM departments 
LEFT OUTER JOIN hod ON (departments.did = hod.hid) 
ORDER BY did ASC;

SELECT * FROM departments FULL JOIN Hod ON (departments.did = Hod.hid);
